- `reflect`パッケージの`TypeOf`を使います。

```:main.go
package main

import (
    "fmt"
    {em{"reflect"}em}
)

func main() {
    n := 10
    var n16 int16 = 10
    fmt.Println(n, {em{reflect.TypeOf(}em}n{em{)}em})
    fmt.Println(n16, {em{reflect.TypeOf(}em}n16{em{)}em})
}
```

```output:出力結果
10 int
10 int16
```
