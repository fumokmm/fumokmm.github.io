```:build.gradle
apply from: 'script1.gradle'
```

```:script1.gradle
buildscript {
    repositories {
        mavenCentral()
    }
    dependencies {
        classpath 'org.apache.groovy:groovy-all:4.0.8'
        classpath 'org.apache.groovy:groovy-yaml:4.0.8'
    }
}

import groovy.yaml.*
task foo {
    doFirst {
        println("hello")
        {em:comment{// ここでYamlSlurperが使える！}em}
        def yaml = new YamlSlurper()
        def data = yaml.parseText('''
        key1: value1
        key2: value2
        ''')
        println yaml {em:comment{// groovy.yaml.YamlSlurper@xxxxxxxx}em}
        println(data.key1) {em:comment{// Output: value1}em}
    }
}
```
